 ```    
 kNN <- function(xl, z, k)
{
  ###### –°–æ—Ä—Ç–∏—Ä—É–µ–º –≤—ã–±–æ—Ä–∫—É —Å–æ–≥–ª–∞—Å–Ω–æ –∫–ª–∞—Å—Å–∏—Ñ–∏—Ü–∏—Ä—É–µ–º–æ–≥–æ

  n <- dim(orderedXl)[2] - 1
  ###### –ü–æ–ª—É—á–∞–µ–º –∫–ª–∞—Å—Å—ã –ø–µ—Ä–≤—ã—Ö k —Å–æ—Å–µ–¥–µ–π
  classes <- orderedXl[1:k, n + 1]
  ###### –°–æ—Å—Ç–∞–≤–ª—è–µ–º —Ç–∞–±–ª–∏—Ü—É –≤—Å—Ç—Ä–µ—á–∞–µ–º–æ—Å—Ç–∏ –∫–∞–∂–¥–æ–≥–æ –∫–ª–∞—Å—Å–∞
  counts <- table(classes)
  ###### –ù–∞—Ö–æ–¥–∏–º –∫–ª–∞—Å—Å, –∫–æ—Ç–æ—Ä—ã–π –¥–æ–º–∏–Ω–∏—Ä—É–µ—Ç —Å—Ä–µ–¥–∏ –ø–µ—Ä–≤—ã—Ö k —Å–æ—Å–µ–¥–µ–π
  class <- names(which.max(counts))
  return (class)
}
```

```
PW = function(XL,y,h,metricFunction = euclideanDistance){    
    l <- dim(xl)[1]  
    weights = rep(0,3)  
    names(weights) = unique(xl[,3])  
    for(i in 1:l)  
    {  
      x=XL[i,1:2]  
      class=XL[i,3]  
      r = metricFunction(x,y)/h  
      weights[class]=kernelR(r)+weights[class];  
    }  
    if(max(weights)==0){ return ("0") }  
    else{ return (names(which.max(weights))) }  
}  
```
